function buildRawDataDetailHTML(rawData, dataId) { // 调试信息 var debugInfo = 'rawData type: ' + typeof rawData + ', keys: ' + Object.keys(rawData).join(', '); let html = '
'; // 标题 let title = '未命名'; if (rawData.title) { title = rawData.title; } html += '
' + escapeHtml(title) + '
'; // 调试信息展示 html += '
'; html += '调试: ' + escapeHtml(debugInfo); html += '
'; // 元信息 html += '
'; html += '
元信息
'; html += '
'; html += '
_meta_id' + escapeHtml(String(dataId)) + '
'; html += '
_id' + escapeHtml(String(rawData._id || '')) + '
'; html += '
'; // 原始数据 JSON html += '
'; html += '
原始数据 (JSON)
'; try { var jsonStr = JSON.stringify(rawData, null, 2); html += '
' + escapeHtml(jsonStr) + '
'; } catch (e) { html += '
JSON 序列化失败: ' + escapeHtml(String(e)) + '
'; } html += '
'; html += '
'; return html; }
